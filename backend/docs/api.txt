1.5.1 Create transaction (payer)
╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌╌

  ┌────
  │ email_src_trn = POST('/api/wallet/transactions', {
  │     "to_email": "a@b.cz",
  │     "msatoshi": 1000,
  │     "description": "Free BTC",    # optional - visible to both payee & payer
  │     "expires_after": 900          # in seconds
  │ })['data']
  └────

  ┌────
  │ 201 Created
  │ 
  │ {
  │   "data": {
  │     "claim_expires_at": "2018-08-27T07:04:09.506776",
  │     "description": "Free BTC",
  │     "id": "a923af54-d7e7-480a-9f9c-510e438c46fa",
  │     "inserted_at": "2018-08-27T06:49:09.506849",
  │     "msatoshi": -1000,
  │     "processed_at": null,
  │     "state": "initial"
  │   }
  │ }
  └────


◊ 1.5.1.1 With invalid amount

  ┌────
  │ _ = POST('/api/wallet/transactions', {
  │     "to_email": "a@b.cz",
  │     "msatoshi": -1000, # <- can't send negative amount
  │     "description": "Free BTC",
  │     "expires_after": 900
  │ })
  └────

  ┌────
  │ 400 Bad Request
  │ 
  │ {
  │   "error": {
  │     "detail": "Non-positive amount given"
  │   }
  │ }
  └────
