◊ 1.5.1.1 With amount exceeding wallet balance

  It returns declined transaction.

  ┌────
  │ wallet = silent(lambda: GET('/api/wallet'))
  │ _ = POST('/api/wallet/transactions', {
  │     "to_email": "a@b.cz",
  │     "msatoshi": wallet['data']['balance']['msatoshi'] + 1,
  │     "description": "Free BTC",
  │     "expires_after": 900
  │ })
  └────

  ┌────
  │ 201 Created
  │ 
  │ {
  │   "data": {
  │     "claim_expires_at": "2018-08-28T21:52:03.209147",
  │     "description": "Free BTC",
  │     "id": "574b6454-fb8f-4f0b-b45b-ca5c56d07a52",
  │     "inserted_at": "2018-08-28T21:37:03.209264",
  │     "msatoshi": -999850001,
  │     "processed_at": "2018-08-28T21:37:03.219360",
  │     "state": "declined"
  │   }
  │ }
  └────
